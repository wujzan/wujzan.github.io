import java.util.Scanner;

public class problemA {

	public static final int MAX=10000001; 
	public static boolean[] prime=new boolean[MAX];
	
	public static void main(String[] args) {
		Scanner sc=new Scanner(System.in);
	
		prime[0]=true;		//因boolean預設為false,把非質數設為true
		
		int i,num=2;		//本題題目特殊 "1"其實不是質數
		while(num<MAX) {	//Eratosthenes建立質數表
			if(!prime[num]) {
				for(i=num+num;i<MAX;i+=num) {	//若用i=num*num 超過int 會溢位
					if(prime[i]) continue;
					prime[i]=true;				
				}
			}
			num++;
		}
		
		while(sc.hasNext()) {
			int N=sc.nextInt();
			int C=sc.nextInt();
			int count=0;
			for(i=N;i<=C;i++) {
				if(!prime[i])	
					if( isDPrime(i) ) {
						count++;
						
						if (count > 1)	System.out.print(" ");
						
						System.out.print(i);
					}
			}
			System.out.println();
		}		
	}
	
	public static boolean isDPrime(int n) {
		int z=1;
		int reversed=0;
		while(n>0) {
			int temp=n%10;
			n/=10;
			if( temp==3 || temp==4 || temp==7 ) 
				return false;
			else if(temp==2) 
				temp=5;
			else if(temp==5) 
				temp=2;
			else if(temp==6) 
				temp=9;
			else if(temp==9) 
				temp=6;
		
			reversed+=temp*z;
			z*=10;
		}		
		return( !prime[reversed] );
	}
}

import java.util.Scanner;
import java.util.Stack;

public class problemB {
	public static void main(String[] args) {
		Scanner sc=new Scanner(System.in);
		int n=Integer.parseInt(sc.nextLine());
		while(n--!=0) {		
			//Stack<Character> stack=new Stack<Character>();
			Stack<Integer> stack=new Stack<Integer>();
			String str=sc.nextLine();
			char[] ch=str.toCharArray();
			for(int i=0;i<ch.length;i++) {
			
				if(ch[i]>='0' && ch[i]<='9') {
					stack.push(ch[i]-'0');
				}else if(ch[i]=='+') {
					int b=stack.peek();
					stack.pop();
					int a=stack.peek();
					stack.pop();
					int count=a+b;
					stack.push(count);
				}else if(ch[i]=='-') {
					int b=stack.peek();
					stack.pop();
					int a=stack.peek();
					stack.pop();
					int count=a-b;
					stack.push(count);
				}else if(ch[i]=='*') {
					int b=stack.peek();
					stack.pop();
					int a=stack.peek();
					stack.pop();
					int count=a*b;
					stack.push(count);
				}else if(ch[i]=='/') {
					int b=stack.peek();
					stack.pop();
					int a=stack.peek();
					stack.pop();
					int count=a/b;
					stack.push(count);
				}
			}
			System.out.println(stack.peek());
		}
	}

}

import java.util.Scanner;
import java.util.Stack;

public class problemC {
	public static int[] cookie;
	
	public static void main(String[] args) {
		Scanner sc=new Scanner(System.in);
		
		while(sc.hasNext()) {
			char[] str=sc.nextLine().toCharArray();
			cookie=new int[str.length];
			for(int i=0;i<str.length;i++)
				cookie[i]=(str[i]-'A')+1;	//介於1~26
		
			for(int j=0;j<cookie.length;j++)
				if(j!=cookie.length-1)
					System.out.print((char) (cookie[j]-1+'A')+" ");
				else
					System.out.print((char) (cookie[j]-1+'A'));
			System.out.println();
			
			int down=cookie.length-1;
			while(down!=0) {  //(超過3秒 超時)
				int Max=-1;
				int pos=-1;
				for(int j=down;j>=0;j--)
					if(Max<cookie[j]) {
						Max=cookie[j];
						pos=j;
					}
				
				if(pos==0) {	//最大的在頂層
					flip(cookie,down);
					System.out.print(cookie.length-down+" ");
				}else if(pos==down) {	//最大的在底層(不含已排序對應位置)
					down--;					
				}else {
					flip(cookie,pos);
					System.out.print(cookie.length-pos+" ");
				}			
								
			}
			System.out.println("0");
			//for(int i=0;i<cookie.length;i++) System.out.print(cookie[i]);	
		}		
	}
	
	public static void flip(int[] cookie,int down) {
		 for(int i=0, j=down ; i<j; cookie[i]=cookie[i]^cookie[j], cookie[j]=cookie[i]^cookie[j], cookie[i]=cookie[i++]^cookie[j--]);

import java.util.Scanner;
import java.util.Stack;

public class problemD {
	public static int max=-1;
	
	public static void main(String[] args) {
		Scanner sc=new Scanner(System.in);
		while(sc.hasNext()) {
			max=-1;
			Stack<Character> stack=new Stack<Character>();
			String str=sc.nextLine();
			if(str.length()==0) break;
			
			boolean flag=false;
			String str2="";
			for(int i=0;i<str.length();i++) {
				if(str.charAt(i)=='(' || str.charAt(i)==')') {
					str2+=str.charAt(i);
					flag=true;
				}
			}
					
			
			if(balanced(str) && flag) {			
				System.out.println(max);
			}else {				
				System.out.println(0);
			}		
			
		}		
	}
	
	public static boolean balanced(String line){		
		char[] ch=line.toCharArray();

		Stack<Character> stack=new Stack<Character>();
		for(int i=0;i<ch.length;i++){
			if(ch[i]=='(') {
				stack.push('(');
				if(max<stack.size())
					max=stack.size();
			}else if(ch[i]==')') {
				if(stack.isEmpty())	return false;
				if(stack.peek()=='('){
					stack.pop();
				}else 	
					return false;				
			}		

		}
		return stack.isEmpty();
	}	
}

import java.util.Scanner;

public class problemE {	//給 前序和中序 轉為 後序式(postfix)
	static String pre,in;
	public static void main(String[] args) {
		Scanner sc=new Scanner(System.in);
		
		while(sc.hasNext()) {
			pre=sc.next();
			in=sc.next();
			int len=pre.length();
		
			buildtree(0,len,0);		
		}
	}
	
	public static void buildtree(int start,int ends,int pre_idx) {
		if(start>=ends) return;
		
		int pos=-1;
		
		for(int i=start;i<ends;i++) {			
			if(pre.charAt(pre_idx)==in.charAt(i))  pos=i;	//if(pre[pre_idx]==in[i]) pos=i;
		}
		
		int left=pos-start;
		buildtree(start,start+left,pre_idx+1);
		buildtree(pos+1,ends,pre_idx+left+1);
		
		System.out.print(in.charAt(pos));	//System.out.print(in[pos]);
	}
}
