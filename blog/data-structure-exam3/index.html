import java.util.Scanner;
import java.util.Vector;

public class Pa {
	public static void main(String[] args) {
		Scanner sc=new Scanner(System.in);
		
		// 題目寫錯範圍1<=N<=10000000，實際測資沒那麼大
		int i,MAX=18500000;    
		boolean[] prime=new boolean[MAX];
		
		prime[0]=true;
		prime[1]=true;
		for(i=2;i<MAX;i++) {
			if(!prime[i]) {
				for(int j=i+i;j<MAX;j+=i) {
					if(prime[j]==true) continue;
					prime[j]=true;
				}
			}
		}
		
		int[] ans=new int[15000001];
		int p=1;
		for(i=2;i<MAX;i++) {
			if(!prime[i]) ans[p++]=i;
		}
		
		//System.out.println(p);
		
		while(sc.hasNext()){
			int n=sc.nextInt();		
			System.out.println(ans[n]);
		}
	}
}

import java.util.Scanner;
import java.util.Stack;

public class Pb {
	public static void main(String[] args) {		
		Scanner sc=new Scanner(System.in);
				
		while(sc.hasNext()) {
			int n=Integer.parseInt(sc.nextLine());
			int max=-99999999;
			
			
			for(int cases=0;cases<n;cases++) {
				Stack<Integer> stack=new Stack<Integer>();
				String str=sc.nextLine();
				char[] ch=str.toCharArray();
			
				for(int i=0;i<ch.length;i++) {
			
					if(ch[i]>='0' && ch[i]<='9') {
						stack.push(ch[i]-'0');
					}else if(ch[i]=='+') {
						int b=stack.peek();
						stack.pop();
						int a=stack.peek();
						stack.pop();
					
						int count=a+b;
						stack.push(count);
					}else if(ch[i]=='-') {
						int b=stack.peek();
						stack.pop();
						int a=stack.peek();
						stack.pop();
					
						int count=a-b;
						stack.push(count);
					}else if(ch[i]=='*') {
						int b=stack.peek();
						stack.pop();
						int a=stack.peek();
						stack.pop();
					
						int count=a*b;
						stack.push(count);
					}else if(ch[i]=='/') {
						int b=stack.peek();
						stack.pop();
						int a=stack.peek();
						stack.pop();
					
						int count=a/b;
						stack.push(count);
					}
				}
				
				if(max<stack.peek()) max=stack.peek();
			}			
			
			System.out.println(max);
		}
	}

}

Java程式碼
import java.util.Scanner;
import java.util.Stack;

public class Pd {
	public static int max=-1;
	public static int max2=-1;
	public static void main(String[] args) {
		Scanner sc=new Scanner(System.in);
		while(sc.hasNext()) {
			
			Stack<Character> stack=new Stack<Character>();
			String str=sc.nextLine();
			
			String strA="";
			String strB="";
			
			for(int i=0;i<str.length();i++){
				if(str.charAt(i)=='(' || str.charAt(i)==')')
					strA+=str.charAt(i);
				else
					strB+=str.charAt(i);
			}
			
			max=-1;
			max2=-1;
			
			if(isBalance(str)) {
				max=-1;
				max2=-1;
				isBalance(strA);
				isBalance(strB);
				System.out.println(Math.max(max2, max));
			}else {				
				System.out.println("0");
			}		
			
		}		
	}
	
	public static boolean isBalance(String line){		
		char[] ch=line.toCharArray();
		Stack<Character> stack=new Stack<Character>();
				
		for(int i=0;i<ch.length;i++){
			if(ch[i]=='(') {
				stack.push('(');
				if(max<stack.size())	max=stack.size();
			}else if(ch[i]=='[') {
				stack.push('[');
				if(max2<stack.size())	max2=stack.size();
			}else if(ch[i]==')') {
				if(stack.isEmpty())	return false;
				if(stack.peek()=='('){
					stack.pop();
				}else
					return false;				
			}else if(ch[i]==']') {
				if(stack.isEmpty())	return false;
				if(stack.peek()=='['){
					stack.pop();
				}else
					return false;
			}			
		}
		return stack.isEmpty();
	}
}

import java.util.Scanner;

public class Pe {

	public static void main(String[] args) {
		Scanner sc=new Scanner(System.in);
		
		while(sc.hasNext()) {			
			long n=sc.nextInt();
			if(n==0) break;
			
			while(n>9) {
				n=n%10+n/10;
			}
			System.out.println(n);
		}
	}

}
