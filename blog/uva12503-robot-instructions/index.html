[UVA12503] Robot Instructions

中文翻譯

機器人,輸入一串移動指令,問最後位置

Java程式碼
import java.util.*;

public class main{
  public static void main(String[] args) {
    Scanner sc=new Scanner(System.in);
    int cases=sc.nextInt();
    while(cases--!=0){
    	int n=sc.nextInt();
    	int[] arr=new int[n];
    	sc.nextLine();	//注意
    	int sum=0;
    	int i=0;
		
    	while(n--!=0){
    		String s=sc.nextLine();

    		if(s.equals("LEFT"))
    			arr[i]=-1;
    		else if(s.equals("RIGHT"))
    			arr[i]=1;
    		else{	//s.contains("SAME AS")
    			s=s.substring(8,s.length());
    			arr[i]=arr[Integer.parseInt(s)-1];
    		}
    		i++;
    	}
		
    	for(int j=0;j<arr.length;j++)
    		sum+=arr[j];
    	System.out.println(sum);
    }
  }
};
