倒水問題 BFS

輸入A,B,C三杯子的容量，A,B為空杯子,C為裝滿水

後相倒水，倒到自己倒空或是別人杯子倒滿為止

如無法做到恰好D升, 就倒D’升, D'<D儘量接近

2019-03-26 CPE 第七題 ,官方code

由CPE官方C++解答改寫成Java

Java程式碼
import java.util.*;

public class Main{

  static int A, B, C, D;	
  static int[][] water=new int[201][201];	//三杯水的總和固定,開二維陣列即可
  static int[] ans=new int[201];	//不大於200升 --> 0~200
	
  static Queue<Integer> queuea = new LinkedList<Integer>();
  static Queue<Integer> queueb = new LinkedList<Integer>();
  static Queue<Integer> queuec = new LinkedList<Integer>();
  static Queue<Integer> queuetotal = new LinkedList<Integer>();

  public static void Main(String[] args) { 	 //uva10603
    Scanner sc=new Scanner(System.in);
    int T=sc.nextInt();
    while(T--!=0){
    	int n, i, j, k;
    
    	A=sc.nextInt();
    	B=sc.nextInt();
    	C=sc.nextInt();
		D=sc.nextInt();
    
    	for (i = 0; i <= A; i++)
    		for (j = 0; j <= B; j++)
        		water[i][j] = Integer.MAX_VALUE;
				
    	for (i = 0; i <= D; i++)
        	ans[i] = Integer.MAX_VALUE;
			
   		bfs(0, 0, C, 0);
		
    	for (i = D; ans[i] == Integer.MAX_VALUE ; i--);
    
    	System.out.println(ans[i]+" "+i);	
		//ans[i]-->最小倒水量	i-->目標水量
    }
  }
  
  public static void bfs(int a, int b, int c, int total){
    pushnode(a, b, c, total);
    while (!queuea.isEmpty()){
    	a = queuea.remove();    	
        b = queueb.remove();
        c = queuec.remove();
        total = queuetotal.remove();
        update(a, b, c, total);
    }
  }
  
  public static void pushnode(int a, int b, int c, int total){
    queuea.add(a); queueb.add(b); queuec.add(c); queuetotal.add(total);
  }

  public static void update(int a, int b, int c, int total){
    if (total >= ans[D]) return;
    if (total >= water[a][b]) return;
    water[a][b] = total;
    ans[a] = Math.min(ans[a], total);
    ans[b] = Math.min(ans[b], total);
    ans[c] = Math.min(ans[c], total);
	
	/*
	大寫"C"杯子容量上限
	小寫"c"目前杯中水量
	杯子"C-c"還能倒的容量
	*/
	
    if (a <= C - c)
        pushnode(0, b, c+a, total+a);
    else
        pushnode(a-C+c, b, C, total+C-c);
    if (a <= B - b)
        pushnode(0, b+a, c, total+a);
    else
        pushnode(a-B+b, B, c, total+B-b);
    if (b <= C - c)
        pushnode(a, 0, c+b, total+b);
    else
        pushnode(a, b-C+c, C, total+C-c);
    if (b <= A - a)
        pushnode(a+b, 0, c, total+b);
    else
        pushnode(A, b-A+a, c, total+A-a);
    if (c <= A - a)
        pushnode(a+c, b, 0, total+c);
    else
        pushnode(A, b, c-A+a, total+A-a);
    if (c <= B - b)
        pushnode(a, b+c, 0, total+c);
    else
        pushnode(a, B, c-B+b, total+B-b);
  }  
  
};

Java程式碼
import java.util.Arrays;
import java.util.Comparator;
import java.util.PriorityQueue;
import java.util.Scanner;

class node {
	int[] v=new int[3];
	int vol;

	node() {}
	node(node n){
		this.vol=n.vol;
		
		this.v[0]=n.v[0];
		this.v[1]=n.v[1];
		this.v[2]=n.v[2];
	}
		
}

class toSortTemplate implements Comparator<node>{
	public int compare(node o1,node o2){
		if(o1.vol>o2.vol) return 1;
		if(o1.vol<o2.vol) return -1;
		return 0;		 
	}
}

public class Main {		//uva10603 狀態空間搜索(狀態樹)
	public static int maxn=201,a,b,c,d;
	public static boolean[][] vis=new boolean[maxn][maxn];
	public static int[] ans=new int[maxn];
	public static int[] cap=new int[3];
	
	//public static Queue<node> Q=new LinkedList<node>();  //若用Queue CPE測資(太弱)過，但OJ Wrong answer
	public static PriorityQueue<node> Q=new PriorityQueue<node>(new toSortTemplate());
	//為了保證vol(倒水量)最小，可以把狀態扔進以vol從小到大排序的PriorityQueue裡，此時每次取出的都很可能是當前vol對應的倒水量最少的狀態， 用它更新vol對應的答案。直到結束。
	
	public static void main(String[] args) {
		Scanner sc=new Scanner(System.in);

		int T=sc.nextInt();
		while(T--!=0) {
			a=sc.nextInt();
			b=sc.nextInt();
			c=sc.nextInt();
			d=sc.nextInt();
			
			for(int i=0;i<maxn;i++)
				for(int j=0;j<maxn;j++)
					vis[i][j]=false;
			
			Arrays.fill(ans, -1);
						
			while(!Q.isEmpty()) Q.remove(); //或不用while直接Q.clear();
			
			node s=new node();
			s.v[0]=0;s.v[1]=0;s.v[2]=c;s.vol=0;	//一開始杯中水量，c杯是裝滿的
			
			cap[0]=a;cap[1]=b;cap[2]=c;	//三杯容量上限
						
			bfs(s);
		}
	}
	
	public static void update_ans(node u) {
		for(int i=0;i<3;i++) {
			int d=u.v[i];
			if(ans[d]<0)  // 不需要 "|| ans[d]>u.vol)"，因為PriorityQueue就是先取出最小的
				ans[d]=u.vol;
			//就算用Queue加上"ans[d]>u.vol"，排列組合的水量update_ans後會是錯的。因為先排了大的vol，update_ans(假如之後v[1]較小，取代)不會是在同一種狀態下
		}
	}
	
	public static void bfs(node start) {
		vis[0][0]=true;	
		Q.add(start);
		
		while(!Q.isEmpty()) {
			node e=Q.peek();
			Q.remove();  //.remove() java函數庫引用的其實是.poll()
			
			vis[e.v[0]][e.v[1]]=true;
			update_ans(e);
			
			if(ans[d]>=0) break;
			
			for(int i=0;i<3;i++) {	//倒水的
				for(int j=0;j<3;j++) {	//被倒入的
					
					//if (i == j) continue; //對程式速度沒影響?????
					if(e.v[i]==0 || e.v[j]==cap[j]) continue;
					
					int amount=Math.min(cap[j], e.v[i]+e.v[j])-e.v[j];	//"-e.v[j]" 減掉被倒原杯中水量。
					//"amount":倒出到另一杯(變動量)。 "Math.min(cap[j]":不超過被倒杯子容量上限。 "e.v[i]+e.v[j]":若全倒過去。
					
					node s=new node(e);	//若用複製的會牽扯到 "淺拷貝" "深拷貝"						
					
					s.vol+=amount;
					s.v[i]-=amount;
					s.v[j]+=amount;				
				
					if(!vis[s.v[0]][s.v[1]]) Q.add(s);					
				}
			}
		}
		while(d>=0) {
			if(ans[d]>=0) {
				System.out.println(ans[d]+" "+d);
				return;
			}
			d--;
		}
	}
	
}
